# 使用

import redis;   //导入redis模块
var value = 'hello';
//通过redis.命令名(命令参数,命令参数,.....,命令参数) 进行调用，其中命令名不区分大小写
redis.set('key',value); //调用set命令
redis.setex('key',10,value);    //调用setex命令
return redis.get('key');    //调用get命令
